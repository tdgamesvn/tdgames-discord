# Config UI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a standalone Express web server (`tools/config-ui/server.ts`) that serves a dark-themed HTML form for reading and writing the Discord bot's `.env` file, accessible via Tailscale on port 3456.

**Architecture:** Single-file Express server with 4 routes. `readEnv`/`writeEnv` helper functions are exported for unit testing. HTML page is inlined as a template string — no build step, no bundler. Bot restart uses PID file written by `src/index.ts` on startup.

**Tech Stack:** Express 4, TypeScript, tsx (already installed), vitest (already installed)

## Global Constraints

- Runner: `tsx` — NOT `ts-node`
- Dev bot script: `npm run dev` → `tsx watch src/index.ts`
- All paths relative to project root (`tdgames-discord/`)
- Port default: `3456`, bind: `0.0.0.0`
- PID file: `data/bot.pid`
- Atomic .env write: write `.env.tmp`, then `fs.renameSync` to `.env`
- No authentication — Tailscale provides network-level security

```typescript
import * as fs from 'fs';
import * as path from 'path';

// ─── .env helpers ─────────────────────────────────────────────────────────────

export function readEnv(filePath: string): Record<string, string> {
  try {
    const content = fs.readFileSync(filePath, 'utf8');
    const result: Record<string, string> = {};
    for (const line of content.split('\n')) {
      const trimmed = line.trim();
      if (!trimmed || trimmed.startsWith('#')) continue;
      const eqIdx = trimmed.indexOf('=');
      if (eqIdx === -1) continue;
      const key = trimmed.slice(0, eqIdx).trim();
      const value = trimmed.slice(eqIdx + 1);
      result[key] = value;
    }
    return result;
  } catch {
    return {};
  }
}

export function writeEnv(
  filePath: string,
  examplePath: string,
  values: Record<string, string>
): void {
  let exampleLines: string[] = [];
  try {
    exampleLines = fs.readFileSync(examplePath, 'utf8').split('\n');
  } catch {
    exampleLines = [];
  }

  const usedKeys = new Set<string>();
  const lines: string[] = [];

  for (const line of exampleLines) {
    const trimmed = line.trim();
    if (!trimmed || trimmed.startsWith('#')) {
      lines.push(line);
      continue;
    }
    const eqIdx = trimmed.indexOf('=');
    if (eqIdx === -1) {
      lines.push(line);
      continue;
    }
    const key = trimmed.slice(0, eqIdx).trim();
    usedKeys.add(key);
    const newValue = key in values ? values[key] : '';
    lines.push(`${key}=${newValue}`);
  }

  // Append keys not in example
  for (const [key, value] of Object.entries(values)) {
    if (!usedKeys.has(key)) {
      lines.push(`${key}=${value}`);
    }
  }

  const tmpPath = filePath + '.tmp';
  fs.writeFileSync(tmpPath, lines.join('\n'), 'utf8');
  fs.renameSync(tmpPath, filePath);
}
```
